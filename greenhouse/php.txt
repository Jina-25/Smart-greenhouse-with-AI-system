-- 1. Reset the live monitoring table
DROP TABLE IF EXISTS `current_plant`;
CREATE TABLE `current_plant` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `plant_name` varchar(50) DEFAULT 'None',
  `temperature` float DEFAULT 0,
  `humidity` float DEFAULT 0,
  `soil_moisture` float DEFAULT 0,
  `light` float DEFAULT 0,
  `timestamp` timestamp DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`)
);

-- 3. Create the optimal settings table
CREATE TABLE IF NOT EXISTS `plants_data` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `plant_name` varchar(50) NOT NULL UNIQUE,
  `temperature` float DEFAULT NULL,
  `humidity` float DEFAULT NULL,
  `soil_moisture` float DEFAULT NULL,
  `light` float DEFAULT NULL,
  `recorded_at` timestamp DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`)
);
